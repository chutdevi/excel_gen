CREATE VIEW v_pemohon_hibah AS
SELECT 
	h.hib_kode as reg_kode,
	h.hib_tanggal as tgl_daftar,
	ho.ho_tanggal as tgl_evaluasi_opd,
	o.opd_nama as opd_tujuan,
	jh.jh_jenis as jenis_hibah,
	h.hib_judul_kegiatan as judul_kegiatan,
	h.hib_lokasi_kegiatan as lokasi_kegiatan,
	h.hib_nphd as no_nphd,
	h.hib_nphd_tgl as tgl_nphd,
	h.hib_nama as pemohon,
	CONCAT(`hib_jalan`,' RT.',`hib_rt`,' / RW.',`hib_rw`) as alamat,
	kel.nm_kelurahan as kelurahan,
	kec.nm_kecamatan as kecamatan,
	dt.nm_dati2 as kota,
	prop.nm_propinsi as propinsi,
	h.hib_kodepos as kodepos,
	h.hib_tlp as no_tlp,
	h.hib_hp as no_hp,
	b.bank_nama as rekening,
	h.hib_norek as no_rekening,
	h.hib_rencana_penggunaan as rencana_penggunaan,
	h.hib_status as status_permohonan,
	h.hib_eval_opd as status_eval_opd,
	h.hib_eval_tapd as status_eval_tapd,
	h.hib_besaran_hibah as besaran_hibah,
	ho.ho_besaran_hibah as besaran_opd,
	h.opd_kode as kode_opd,
	h.jh_kode as kode_jh,
	h.bank_kode as kode_bank,
	h.kd_kelurahan as kode_wilayah,
	ho.ho_id as eval_opd_kode 
FROM tbl_hibah h, tbl_hibah_opd ho, tbl_kelurahan kel, tbl_kecamatan kec, tbl_dati2 dt, tbl_propinsi prop, tbl_jenis_hibah jh, tbl_bank b, tbl_opd o
WHERE h.hib_kode=ho.hib_kode AND h.kd_kelurahan=kel.kd_kelurahan AND h.kd_kecamatan=kec.kd_kecamatan AND h.kd_dati2=dt.kd_dati2 AND h.kd_propinsi=prop.kd_propinsi AND h.jh_kode=jh.jh_kode AND h.bank_kode=b.bank_kode AND h.opd_kode=o.opd_kode
